    <script>
        let yesSize = 1;

        /**
         * Logic for the 'No' button: 
         * Moves it smoothly within a 40px radius and grows the 'Yes' button.
         */
        function handleNo() {
            const noBtn = document.getElementById('noBtn');
            const yesBtn = document.getElementById('yesBtn');

            // Switch to absolute positioning on first interaction
            if (noBtn.style.position !== 'absolute') {
                noBtn.style.position = 'absolute';
            }

            // Calculate current position
            const currentTop = noBtn.offsetTop;
            const currentLeft = noBtn.offsetLeft;

            // Calculate relative random move within 40px radius
            let newTop = currentTop + (Math.random() * 240 - 120);
            let newLeft = currentLeft + (Math.random() * 240 - 120);

            // Boundary checks to keep the button on screen
            const pad = 60;
            newTop = Math.max(pad, Math.min(newTop, window.innerHeight - pad));
            newLeft = Math.max(pad, Math.min(newLeft, window.innerWidth - pad));

            noBtn.style.top = newTop + 'px';
            noBtn.style.left = newLeft + 'px';

            // Make the 'Yes' button grow
            yesSize += 0.15;
            yesBtn.style.transform = `scale(${yesSize})`;
        }

        /**
         * Logic for the 'Yes' button:
         * Switches pages and starts the heart celebration.
         */
        function celebrate() {
            document.getElementById('page1').classList.add('hidden');
            const p2 = document.getElementById('page2');
            p2.classList.remove('hidden');
            
            // Celebration: burst of hearts
            for(let i=0; i<30; i++) {
                setTimeout(createHeart, i * 100);
            }
            // Keep hearts falling
            setInterval(createHeart, 300);
        }

        /**
         * Creates a single animated heart emoji
         */
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('floating-heart');
            const icons = ['‚ù§Ô∏è', 'üíñ', 'üíù', 'üíï', 'üå∏'];
            heart.innerHTML = icons[Math.floor(Math.random() * icons.length)];
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.fontSize = (Math.random() * 20 + 20) + 'px';
            heart.style.animationDuration = (Math.random() * 2 + 3) + 's';
            document.body.appendChild(heart);
            
            // Clean up memory
            setTimeout(() => heart.remove(), 5000);
        }

        // Ambient hearts: start falling slowly on load
        setInterval(createHeart, 1000);
